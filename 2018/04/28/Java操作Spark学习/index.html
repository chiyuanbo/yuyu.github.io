SparkContext sc = new SparkContext(conf);
HiveContext hiveCtx = new HiveContext(sc);
String sql = "select * from fjudm4.hbase_md_load_bus_hisdata limit 1";
Row[] rows = hiveCtx.sql(sql).collect();

for (int i = 0; i < rows.length; i++) {

}
    sc.stop();
